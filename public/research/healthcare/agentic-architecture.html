-- Multi-Vector Patient Similarity with HNSW Index CREATE TABLE patient_embeddings ( patient_id VARCHAR2(20) PRIMARY KEY, diagnosis_vec VECTOR(1536, FLOAT32), -- Cohere Embed 4 morphology_vec VECTOR(512, FLOAT32), -- BiomedCLIP fine-tuned immuno_vec VECTOR(256, FLOAT32), -- Custom MLP encoder molecular_vec VECTOR(256, FLOAT32), -- Mutation signature treatment_vec VECTOR(128, FLOAT32) -- Protocol response ); -- Create HNSW indexes for sub-millisecond search CREATE VECTOR INDEX idx_diagnosis_hnsw ON patient_embeddings(diagnosis_vec) ORGANIZATION INMEMORY NEIGHBOR GRAPH DISTANCE COSINE WITH TARGET ACCURACY 95; -- Weighted multi-vector similarity query SELECT p.patient_id, o.outcome_status, o.event_free_survival_months, ( 0.3 * VECTOR_DISTANCE(p.diagnosis_vec, :query_diag, COSINE) + 0.25 * VECTOR_DISTANCE(p.molecular_vec, :query_mol, COSINE) + 0.25 * VECTOR_DISTANCE(p.morphology_vec, :query_morph, COSINE) + 0.2 * VECTOR_DISTANCE(p.immuno_vec, :query_immuno, COSINE) ) AS weighted_similarity FROM patient_embeddings p JOIN treatment_outcomes o ON p.patient_id = o.patient_id ORDER BY weighted_similarity FETCH FIRST 10 ROWS ONLY;
