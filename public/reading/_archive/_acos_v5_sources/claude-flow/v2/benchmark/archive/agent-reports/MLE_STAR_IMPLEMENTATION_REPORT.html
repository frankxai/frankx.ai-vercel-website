Custom Ensemble

from swarm_benchmark.mle_star import MLEStarEnsembleExecutor, MLEStarConfig

    models=[

        {"type": "gradient_boost", "n_estimators": 100},
        {"type": "neural_network", "layers": [100, 50]}
    ],
    voting_strategy="weighted"
