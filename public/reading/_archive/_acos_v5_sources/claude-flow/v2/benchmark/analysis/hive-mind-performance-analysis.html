  private pools = new Map<string, object[]>();
  
  acquire<T>(type: string): T {
    const pool = this.pools.get(type) || [];
    return pool.pop() || new (this.getConstructor(type))();
  }
  
  release<T>(obj: T, type: string): void {
    this.cleanup(obj);
