// app/api/subscribe/route.ts (already exists!)
import { Resend } from 'resend'
import { NextResponse } from 'next/server'

export async function POST(req: Request) {
  const { email } = await req.json()

  // Add to Resend audience
  const resend = new Resend(process.env.RESEND_API_KEY)
  await resend.contacts.create({
    email,
    audienceId: process.env.RESEND_AUDIENCE_ID,
  })

  // Add to ConvertKit
  await fetch(`https://api.convertkit.com/v3/forms/${process.env.CONVERTKIT_FORM_ID}/subscribe`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      api_key: process.env.CONVERTKIT_API_KEY,
      email,
      tags: ['links-page-signup']
    })
  })

  return NextResponse.json({ success: true })
}

// components/links/VideoBlock.tsx
interface VideoBlockProps {
  platform: 'youtube' | 'vimeo' | 'loom' | 'tiktok' | 'suno'
  videoId: string
  title: string
  description?: string
  thumbnail?: string
}

export function VideoBlock({ platform, videoId, title, description, thumbnail }: VideoBlockProps) {
  const [isPlaying, setIsPlaying] = useState(false)

  const embedUrls = {
    youtube: `https://www.youtube.com/embed/${videoId}`,
    vimeo: `https://player.vimeo.com/video/${videoId}`,
    loom: `https://www.loom.com/embed/${videoId}`,
    tiktok: `https://www.tiktok.com/embed/${videoId}`,
    suno: `https://suno.com/song/${videoId}/embed`
  }

  return (
    <motion.div
      className="rounded-3xl overflow-hidden bg-white/5 backdrop-blur-md border border-white/10 hover:border-white/20 transition-all"
      whileHover={{ scale: 1.02 }}
    >
      {!isPlaying ? (
        // Thumbnail with play button
        <div
          className="relative aspect-video cursor-pointer group"
          onClick={() => setIsPlaying(true)}
        >
          <Image
            src={thumbnail || `/api/og?video=${videoId}`}
            alt={title}
            fill
            className="object-cover"
          />
          <div className="absolute inset-0 bg-gradient-to-t from-black/60 to-transparent" />
          <div className="absolute inset-0 flex items-center justify-center">
            <div className="w-20 h-20 rounded-full bg-white/90 group-hover:bg-white flex items-center justify-center transition-all group-hover:scale-110">
              <Play className="w-8 h-8 text-black ml-1" />
            </div>
          </div>
          <div className="absolute bottom-4 left-4 right-4">
            <h3 className="text-white font-semibold text-lg mb-1">{title}</h3>
            {description && (
              <p className="text-white/80 text-sm">{description}</p>
            )}
          </div>
        </div>
      ) : (
        // Actual video embed
        <div className="aspect-video">
          <iframe
            src={embedUrls[platform]}
            className="w-full h-full"
            allow="accelerometer; autoplay; clipboard-write; encrypted-media; gyroscope; picture-in-picture"
            allowFullScreen
          />
        </div>
      )}
    </motion.div>
  )
}

// lib/link-preview.ts
export async function fetchLinkPreview(url: string) {
  try {
    const res = await fetch(`https://api.microlink.io?url=${encodeURIComponent(url)}`)
    const data = await res.json()

    return {
      title: data.data.title,
      description: data.data.description,
      image: data.data.image?.url,
      favicon: data.data.logo?.url,
      url: data.data.url
    }
  } catch (error) {
    return null
  }
}

// components/links/BlockRenderer.tsx
export function BlockRenderer({ blocks }: { blocks: LinkBlock[] }) {
  return (
    <div className="space-y-6">
      {blocks
        .filter(block => block.visible)
        .sort((a, b) => a.order - b.order)
        .map(block => {
          switch (block.type) {
            case 'video':
              return <VideoBlock key={block.id} {...block.data} />
            case 'spotify':
              return <SpotifyEmbed key={block.id} {...block.data} />
            case 'link':
              return <LinkCard key={block.id} {...block.data} />
            case 'email':
              return <EmailCapture key={block.id} {...block.data} />
            default:
              return null
          }
        })}
    </div>
  )
}

// lib/visibility-rules.ts
interface VisibilityRule {
  type: 'time' | 'location' | 'referrer' | 'device'
  condition: string
  value: any
}

export function checkVisibility(block: LinkBlock): boolean {
  if (!block.rules) return block.visible

  for (const rule of block.rules) {
    switch (rule.type) {
      case 'time':
        const hour = new Date().getHours()
        if (rule.condition === 'between' && (hour < rule.value[0] || hour > rule.value[1])) {
          return false
        }
        break
      case 'referrer':
        if (rule.condition === 'contains' && !document.referrer.includes(rule.value)) {
          return false
        }
        break
      // Add more rules as needed
    }
  }

  return block.visible
}
