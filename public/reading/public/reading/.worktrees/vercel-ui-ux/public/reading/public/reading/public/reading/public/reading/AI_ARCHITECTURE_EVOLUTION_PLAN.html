Implement ARIA Tab Pattern:

function CloudProviderTabs() {
  const [activeProvider, setActiveProvider] = useState<string>('aws')
  const tabsRef = useRef<HTMLDivElement>(null)

  // Keyboard navigation
  const handleKeyDown = (e: React.KeyboardEvent, index: number, keys: string[]) => {
    if (e.key === 'ArrowRight') {
      const nextIndex = (index + 1) % keys.length
      setActiveProvider(keys[nextIndex])
      // Focus next tab button
    } else if (e.key === 'ArrowLeft') {
      const prevIndex = (index - 1 + keys.length) % keys.length
      setActiveProvider(keys[prevIndex])
      // Focus previous tab button
    } else if (e.key === 'Home') {
      setActiveProvider(keys[0])
      // Focus first tab
    } else if (e.key === 'End') {
      setActiveProvider(keys[keys.length - 1])
      // Focus last tab
    }
  }

  return (
    <div role="tablist" aria-label="Cloud Provider Comparison" ref={tabsRef}>
      {Object.keys(cloudProviders).map((key, index, keys) => (
        <button
          key={key}
          role="tab"
          aria-selected={activeProvider === key}
          aria-controls={`panel-${key}`}
          id={`tab-${key}`}
          tabIndex={activeProvider === key ? 0 : -1}
          onKeyDown={(e) => handleKeyDown(e, index, keys)}
          onClick={() => setActiveProvider(key)}
          className={`
            px-6 py-3 rounded-full font-medium transition-all
            focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-emerald-400 focus-visible:ring-offset-2 focus-visible:ring-offset-[#0a0a0b]
            ${activeProvider === key ? 'bg-white text-black' : 'bg-white/5 text-white/70 hover:bg-white/10'}
          `}
        >
          <span className="mr-2" aria-hidden="true">{cloudProviders[key].logo}</span>
          {cloudProviders[key].name}
        </button>
      ))}
    </div>

    <div
      role="tabpanel"
      id={`panel-${activeProvider}`}
      aria-labelledby={`tab-${activeProvider}`}
      className="mt-8"
    >
      {/* Provider details */}
    </div>
  )
}

function CostCalculator() {

    monthlyUsers: 10000,
    storageGB: 100,
    aiApiCalls: 50000,
    computeType: 'serverless',
    databaseType: 'relational',
    region: 'us-east'
  })

  const [results, setResults] = useState<CostEstimate[]>([])

  // Calculate costs for all 4 providers
  const calculate = async () => {
    const estimates = await Promise.all([
      calculateAWSCost(inputs),
      calculateGCPCost(inputs),
      calculateAzureCost(inputs),
      calculateOCICost(inputs),
    ])

    setResults(estimates.sort((a, b) => a.total - b.total))
  }

  return (
    <section className="py-16">
      <h2>AI Cost Calculator</h2>

      <div className="grid md:grid-cols-2 gap-8">
        {/* Input Form */}

              min="1000"
              max="10000000"

import Anthropic from '@anthropic-ai/sdk'

const client = new Anthropic({
  apiKey: process.env.ANTHROPIC_API_KEY
})

export async function POST(req: Request) {
  const { messages, providers } = await req.json()

  const systemPrompt = `You are an expert cloud architecture consultant specializing in AI-powered platforms.

You have deep knowledge of AWS, Google Cloud Platform (GCP), Microsoft Azure, and Oracle Cloud Infrastructure (OCI).

Context: The user is comparing cloud providers for their AI platform. Here's the current comparison data:
${JSON.stringify(providers, null, 2)}

Your task:
1. Understand the user's requirements (traffic, budget, technical stack, team expertise)
2. Recommend the best cloud provider(s) and explain why
3. Suggest specific services to use (compute, database, AI services, etc.)
4. Provide rough cost estimates based on their usage
5. Mention trade-offs and alternatives

Be concise but thorough. Use the comparison data above to give accurate recommendations.`

  const response = await client.messages.create({
    model: 'claude-sonnet-4-5-20250929',
    max_tokens: 1024,
    system: systemPrompt,
    messages: messages.map(m => ({
      role: m.role,
      content: m.content
    }))
  })

  return Response.json({
    message: response.content[0].text
  })
}
