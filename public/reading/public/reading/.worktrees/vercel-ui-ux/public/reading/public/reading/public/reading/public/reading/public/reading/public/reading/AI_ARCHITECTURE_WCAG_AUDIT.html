      10. RECOMMENDED FIX: Tab Component

      function CloudProviderTabs() {
  const [activeProvider, setActiveProvider] = useState<CloudProvider>('aws')
  const [announcement, setAnnouncement] = useState('')

  const handleTabChange = (provider: CloudProvider) => {
    setActiveProvider(provider)
    setAnnouncement(`Now showing ${cloudProviders[provider].name}`)
  }

  const handleKeyDown = (e: React.KeyboardEvent) => {
    const tabs = Object.keys(cloudProviders) as CloudProvider[]
    const idx = tabs.indexOf(activeProvider)

    switch (e.key) {
      case 'ArrowRight':
        e.preventDefault()
        handleTabChange(tabs[(idx + 1) % tabs.length])
        break
      case 'ArrowLeft':
        e.preventDefault()
        handleTabChange(tabs[(idx - 1 + tabs.length) % tabs.length])
        break
      case 'Home':
        e.preventDefault()
        handleTabChange(tabs[0])
        break
      case 'End':
        e.preventDefault()
        handleTabChange(tabs[tabs.length - 1])
        break
    }
  }

  return (
    <section aria-labelledby="comparison-heading">
      {/* Live region */}
      <div role="status" aria-live="polite" className="sr-only">
        {announcement}
      </div>

      <h2 id="comparison-heading">Cloud Provider Comparison</h2>

      {/* Tablist */}
      <div
        role="tablist"
        aria-label="Cloud providers"
        onKeyDown={handleKeyDown}
      >
        {(Object.keys(cloudProviders) as CloudProvider[]).map((key) => (
          <button
            key={key}
            role="tab"
            aria-selected={activeProvider === key}
            aria-controls={`panel-${key}`}
            id={`tab-${key}`}
            tabIndex={activeProvider === key ? 0 : -1}
            onClick={() => handleTabChange(key)}
            className="focus-visible:outline-2 focus-visible:outline-emerald-400"
          >
            <span aria-hidden="true">{cloudProviders[key].logo}</span>
            {cloudProviders[key].name}
          </button>
        ))}
      </div>

      {/* Panel */}
      <div
        role="tabpanel"
        id={`panel-${activeProvider}`}
        aria-labelledby={`tab-${activeProvider}`}
        tabIndex={0}
      >
        {/* Provider content */}
      </div>
    </section>
  )
}
